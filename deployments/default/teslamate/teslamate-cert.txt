apiVersion: cert-manager.io/v1alpha2
kind: Certificate
metadata:
  name: teslamate-cert
  namespace: default
spec:
  # Secret names are always required.
  secretName: teslamate-tls
  duration: 8640h
  renewBefore: 1440h
  organization:
  - teslamate
  isCA: false
  keySize: 4096
  keyAlgorithm: rsa
  keyEncoding: pkcs1
  usages:
    - server auth
    - client auth
  # At least one of a DNS Name, URI, or IP address is required.
  dnsNames:
  - teslamate.$DOMAIN
  - www.teslamate.$DOMAIN
  issuerRef:
    name: ca-issuer
    # We can reference ClusterIssuers by changing the kind here.
    # The default value is Issuer (i.e. a locally namespaced Issuer)
    kind: ClusterIssuer
